@app.route('/api/vulnerable-profile/<user_id>')
def vulnerable_profile(user_id):
    # ❌ 세션 확인 없이 사용자가 넘긴 user_id 데이터 그대로 조회
    profile = db.get_user(user_id)
    return jsonify(profile)
@app.route('/api/secure-profile/<user_id>')
def secure_profile(user_id):
    current_session_user = session.get('user_id')
    if str(current_session_user) != str(user_id):
        # ✅ 로그인된 세션과 요청한 user_id가 다르면 403 차단
        return jsonify({'error': 'Forbidden'}), 403
    return jsonify(db.get_user(user_id))
