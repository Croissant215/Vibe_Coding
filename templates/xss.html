def vulnerable_xss(content):

    return f"<div>검색어: {content}</div>"
@app.route('/api/secure-xss')
def secure_xss(content):
    # ✅ html.escape() 로 특수문자(<, >, &)를 안전한 Entity로 변환
    escaped = html.escape(content)
    return f"<div>검색어: {escaped}</div>"
